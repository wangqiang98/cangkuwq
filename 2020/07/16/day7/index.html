** 一类方法

public class ConditionTest11 {

public static void main(String[] args) {
data(“小A”,4,426);
data(“小B”,4,300);
data(“小C”,4,500);
data(“小D”,6,400);
data(“小E”,6,505);
}
public static void data(String name,int lever, int grade){
if (grade>=425) {
if (lever == 4){
System.out.println(“恭喜你通过四级考试”);
}else if (lever == 6){
System.out.println(“恭喜你通过六级考试”);
}
}
else{ System.out.println(name+”你本次考试的成绩是：”+grade);
}
}
}
